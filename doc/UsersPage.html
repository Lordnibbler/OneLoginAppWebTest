def select_user
  users = all(:xpath, ".//*[@class='js-click-row user']//h4")
  userlist=[]
  users.each do |user|
    userlist.push("#{user.text}")
  end
  selected_user=userlist.sample 
  find(:xpath, ".//*[@class='js-click-row user']//h4[contains(.,'#{selected_user}')]").click
  return selected_user
end
